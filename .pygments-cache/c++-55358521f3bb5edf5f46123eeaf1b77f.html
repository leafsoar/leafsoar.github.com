class LUser: public cocos2d::CCObject{
public:
	CREATE_FUNC(LUser);
	virtual bool init(){
		return true;
	};
	LUser(){
		CCLog("LUser()");
	};
	~LUser(){
		CCLog("LUser().~():%s", m_sUserName.c_str());
	};

	std::string m_sUserName;		// 用户名
	std::string m_sPassword;		// 用户密码

	// 其它数据
	CCSprite* m_pSprite;
};

// 那么我们的使用过程
LUser* lu = LUser::create();
lu->m_sSprite = CCSprite::create("a.png");
// 如果这里不 retain  则以后就用不到了
lu->m_sSprite->retain();
